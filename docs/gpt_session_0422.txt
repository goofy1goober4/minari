// src/renderer/ui/gift/GiftModalRoot.tsx

import React, { useEffect, useMemo } from "react";
import {
  GiftModalRootProps,
  IncomingImageGiftPayload,
  OutgoingKeepsakeGiftPayload,
  KeepsakeDetailPayload,
} from "./giftModal.types";
import { GiftModalShell } from "./GiftModalShell";
import { IncomingGiftContent } from "./IncomingGiftContent";
import { OutgoingKeepsakeContent } from "./OutgoingKeepsakeContent";
import { KeepsakeDetailContent } from "./KeepsakeDetailContent";
import "./GiftModal.css";

export function GiftModalRoot({
  model,
  onAccept,
  onDecline,
  onClose,
  onModalOpened,
  onModalClosed,
}: GiftModalRootProps) {
  // ESC 닫기
  useEffect(() => {
    if (!model.isOpen) return;

    const handleKeyDown = (event: KeyboardEvent) => {
      if (event.key === "Escape" && !model.busy) {
        onClose();
      }
    };

    window.addEventListener("keydown", handleKeyDown);
    return () => window.removeEventListener("keydown", handleKeyDown);
  }, [model.isOpen, model.busy, onClose]);

  // open / close 훅
  useEffect(() => {
    if (model.isOpen && model.mode) {
      onModalOpened?.(model.mode);
    }
    if (!model.isOpen) {
      onModalClosed?.();
    }
  }, [model.isOpen, model.mode, onModalOpened, onModalClosed]);

  const content = useMemo(() => {
    if (!model.payload || !model.mode) return null;

    switch (model.mode) {
      case "incoming-image-gift":
        return (
          <IncomingGiftContent
            payload={model.payload as IncomingImageGiftPayload}
          />
        );

      case "outgoing-keepsake-gift":
        return (
          <OutgoingKeepsakeContent
            payload={model.payload as OutgoingKeepsakeGiftPayload}
          />
        );

      case "keepsake-detail":
        return (
          <KeepsakeDetailContent
            payload={model.payload as KeepsakeDetailPayload}
          />
        );

      default:
        return null;
    }
  }, [model.mode, model.payload]);

  if (!model.isOpen || !model.payload || !model.mode) {
    return null;
  }

  return (
    <div
      className="gift-modal-root"
      aria-modal="true"
      role="dialog"
      onMouseDown={(e) => {
        // 배경 클릭 시 닫기
        if (e.target === e.currentTarget && !model.busy) {
          onClose();
        }
      }}
    >
      <GiftModalShell
        mode={model.mode}
        viewState={model.viewState}
        busy={!!model.busy}
        onAccept={onAccept}
        onDecline={onDecline}
        onClose={onClose}
      >
        {content}
      </GiftModalShell>
    </div>
  );
}

// src/renderer/ui/gift/GiftModalShell.tsx

import React, { ReactNode } from "react";
import { GiftModalMode, GiftModalViewState } from "./giftModal.types";
import { GiftModalActions } from "./GiftModalActions";

interface GiftModalShellProps {
  mode: GiftModalMode;
  viewState: GiftModalViewState;
  busy: boolean;
  children: ReactNode;
  onAccept?: () => Promise<void> | void;
  onDecline?: () => Promise<void> | void;
  onClose: () => void;
}

function getHeaderTitle(mode: GiftModalMode): string {
  switch (mode) {
    case "incoming-image-gift":
      return "for Minari";
    case "outgoing-keepsake-gift":
      return "a small thing";
    case "keepsake-detail":
      return "keepsake";
    default:
      return "";
  }
}

function getFooterHint(mode: GiftModalMode): string {
  switch (mode) {
    case "incoming-image-gift":
      return "Minari wants to keep this nearby.";
    case "outgoing-keepsake-gift":
      return "You can take it or let it wait.";
    case "keepsake-detail":
      return "A small memory, kept quietly.";
    default:
      return "";
  }
}

export function GiftModalShell({
  mode,
  viewState,
  busy,
  children,
  onAccept,
  onDecline,
  onClose,
}: GiftModalShellProps) {
  return (
    <div className={`gift-modal-shell is-${viewState}`}>
      <div className="gift-modal-header">
        <div className="gift-modal-title">{getHeaderTitle(mode)}</div>
        <button
          className="gift-modal-close"
          onClick={onClose}
          disabled={busy}
          aria-label="Close"
          type="button"
        >
          ×
        </button>
      </div>

      <div className="gift-modal-body">{children}</div>

      <GiftModalActions
        mode={mode}
        busy={busy}
        onAccept={onAccept}
        onDecline={onDecline}
        onClose={onClose}
      />

      <div className="gift-modal-footer-hint">{getFooterHint(mode)}</div>
    </div>
  );
}

import { create } from "zustand";
import {
  GiftModalModel,
  GiftModalStore,
  GiftModalViewState,
} from "./giftModal.types";

const initialModel: GiftModalModel = {
  isOpen: false,
  mode: null,
  viewState: "hidden",
  payload: null,
  busy: false,
  errorMessage: null,
};

export const useGiftModalStore = create<GiftModalStore>((set) => ({
  model: initialModel,

  openIncomingImageGift: (payload) =>
    set({
      model: {
        isOpen: true,
        mode: "incoming-image-gift",
        viewState: "awaiting-choice",
        payload,
        busy: false,
        errorMessage: null,
      },
    }),

  openOutgoingKeepsakeGift: (payload) =>
    set({
      model: {
        isOpen: true,
        mode: "outgoing-keepsake-gift",
        viewState: "awaiting-choice",
        payload,
        busy: false,
        errorMessage: null,
      },
    }),

  openKeepsakeDetail: (payload) =>
    set({
      model: {
        isOpen: true,
        mode: "keepsake-detail",
        viewState: "revealed",
        payload,
        busy: false,
        errorMessage: null,
      },
    }),

  setViewState: (viewState: GiftModalViewState) =>
    set((state) => ({
      model: {
        ...state.model,
        viewState,
      },
    })),

  setBusy: (busy: boolean) =>
    set((state) => ({
      model: {
        ...state.model,
        busy,
      },
    })),

  setError: (message: string | null) =>
    set((state) => ({
      model: {
        ...state.model,
        errorMessage: message,

  closeGiftModal: () =>
    set((state) => ({
      model: {
        ...state.model,
        viewState: "closing",
      },
    })),

    }),

// src/renderer/ui/gift/giftModal.controller.ts

import {
  GiftPersistenceApi,
  IncomingImageGiftPayload,
  OutgoingKeepsakeGiftPayload,
} from "./giftModal.types";
import { useGiftModalStore } from "./giftModal.store";

function sleep(ms: number): Promise<void> {
  return new Promise((resolve) => setTimeout(resolve, ms));
}

export interface GiftModalControllerDeps {
  persistenceApi: GiftPersistenceApi;
  onCharacterMoodChange?: (
    mood:
      | "calm"
      | "curious"
      | "shy"
      | "happy"
      | "sleepy"
      | "sulky"
  ) => void;
  onCharacterPoseChange?: (
    pose:
      | "idle_stand"
      | "holding_out_item"
      | "leaning_in"
      | "bag_rummage"
      | "small_bounce"
  ) => void;
}

export function createGiftModalController(
  deps: GiftModalControllerDeps
) {
  const store = useGiftModalStore;

  return {
    openIncomingImageGift(payload: IncomingImageGiftPayload) {
      store.getState().openIncomingImageGift(payload);

      deps.onCharacterMoodChange?.("happy");
      deps.onCharacterPoseChange?.("leaning_in");
    },

    openOutgoingKeepsakeGift(payload: OutgoingKeepsakeGiftPayload) {
      store.getState().openOutgoingKeepsakeGift(payload);

      deps.onCharacterMoodChange?.("shy");
      deps.onCharacterPoseChange?.("holding_out_item");
    },

    openKeepsakeDetail(keepsakeId: string, title: string, summary: string) {
      store.getState().openKeepsakeDetail({
        type: "keepsake-detail",
        keepsakeId,
        title,
        summary,
        acquiredAt: new Date().toISOString(),
        source: "gift_from_minari",
      });

      deps.onCharacterMoodChange?.("calm");
      deps.onCharacterPoseChange?.("idle_stand");
    },

    async acceptCurrentGift() {
      const { model, setBusy, setViewState, setError, resetGiftModal } =
        store.getState();

      if (!model.payload || !model.mode) return;
      if (model.busy) return;

      try {
        setBusy(true);
        setError(null);
        setViewState("accepting");

        if (model.payload.type === "incoming-image-gift") {
          deps.onCharacterMoodChange?.("happy");
          deps.onCharacterPoseChange?.("small_bounce");

          await deps.persistenceApi.saveIncomingImageGift(model.payload);
        } else if (model.payload.type === "outgoing-keepsake-gift") {
          deps.onCharacterMoodChange?.("happy");
          deps.onCharacterPoseChange?.("holding_out_item");

          await deps.persistenceApi.saveOutgoingKeepsakeAccepted(model.payload);
        } else {
          // detail view에서는 accept 없음
          setBusy(false);
          return;
        }

        setViewState("saved");
        setBusy(false);

        await sleep(450);

        deps.onCharacterMoodChange?.("calm");
        deps.onCharacterPoseChange?.("idle_stand");
        resetGiftModal();
      } catch (error) {
        console.error("[GiftModal] accept failed:", error);
        setBusy(false);
        setError("Could not save this keepsake right now.");

        deps.onCharacterMoodChange?.("sulky");
        deps.onCharacterPoseChange?.("idle_stand");
      }
    },

    async declineCurrentGift() {
      const { model, setBusy, setViewState, setError, resetGiftModal } =
        store.getState();

      if (!model.payload || !model.mode) return;
      if (model.busy) return;

      try {
        setBusy(true);
        setError(null);
        setViewState("declining");

        if (model.payload.type === "outgoing-keepsake-gift") {
          await deps.persistenceApi.saveOutgoingKeepsakeDeclined(model.payload);
        }

        // incoming-image-gift의 later는 저장 안 하고 그냥 닫기
        deps.onCharacterMoodChange?.("shy");
        deps.onCharacterPoseChange?.("idle_stand");

        await sleep(220);

        setBusy(false);
        resetGiftModal();
      } catch (error) {
        console.error("[GiftModal] decline failed:", error);
        setBusy(false);
        setError("Could not close this gift properly.");
      }
    },

    closeModal() {
      const { model, busy, resetGiftModal } = store.getState();

      if (busy) return;

      if (model.isOpen) {
        deps.onCharacterMoodChange?.("calm");
        deps.onCharacterPoseChange?.("idle_stand");
      }

      resetGiftModal();
    },
  };
}

  GIFT_IPC_CHANNELS,

  OutgoingKeepsakeAcceptedSaveRequest,
  OutgoingKeepsakeDeclinedSaveRequest,

function isNonEmptyString(value: unknown): value is string {
  return typeof value === "string" && value.trim().length > 0;
}

function validateIncomingImageGiftRequest(
  payload: unknown
): payload is IncomingImageGiftSaveRequest {
  if (!payload || typeof payload !== "object") return false;
  const p = payload as IncomingImageGiftSaveRequest;

  return (
    isNonEmptyString(p.imagePath) &&
    isNonEmptyString(p.minariReactionLine) &&
    isNonEmptyString(p.visionSummaryShort) &&
    isNonEmptyString(p.createdAtIso)
  );
}

function validateOutgoingAcceptedRequest(
  payload: unknown
): payload is OutgoingKeepsakeAcceptedSaveRequest {
  if (!payload || typeof payload !== "object") return false;
  const p = payload as OutgoingKeepsakeAcceptedSaveRequest;

  return (
    isNonEmptyString(p.keepsakeId) &&
    isNonEmptyString(p.itemKey) &&
    isNonEmptyString(p.displayName) &&
    isNonEmptyString(p.shortLine) &&
    isNonEmptyString(p.acceptedAtIso)
  );
}

function validateOutgoingDeclinedRequest(
  payload: unknown
): payload is OutgoingKeepsakeDeclinedSaveRequest {
  if (!payload || typeof payload !== "object") return false;
  const p = payload as OutgoingKeepsakeDeclinedSaveRequest;

  return (
    isNonEmptyString(p.keepsakeId) &&
    isNonEmptyString(p.itemKey) &&
    isNonEmptyString(p.declinedAtIso)
  );
}

export function registerGiftIpc(): void {
  const repo = new GiftRepository(database);

  ipcMain.handle(
    GIFT_IPC_CHANNELS.saveIncomingImageGift,
    async (_event, payload: unknown): Promise<IpcResult> => {
      try {
        if (!validateIncomingImageGiftRequest(payload)) {
          return { ok: false, error: "Invalid incoming image gift payload." };
        }

        const result = await repo.saveIncomingImageGift(payload);
        return { ok: true, data: result };
      } catch (error) {
        console.error("[IPC] saveIncomingImageGift failed:", error);
        return { ok: false, error: "Failed to save incoming image gift." };
      }
    }
  );

  ipcMain.handle(
    GIFT_IPC_CHANNELS.saveOutgoingKeepsakeAccepted,
    async (_event, payload: unknown): Promise<IpcResult> => {
      try {
        if (!validateOutgoingAcceptedRequest(payload)) {
          return { ok: false, error: "Invalid accepted keepsake payload." };
        }

        const result = await repo.saveOutgoingKeepsakeAccepted(payload);
        return { ok: true, data: result };
      } catch (error) {
        console.error("[IPC] saveOutgoingKeepsakeAccepted failed:", error);
        return { ok: false, error: "Failed to save accepted keepsake." };
      }
    }
  );

  ipcMain.handle(
    GIFT_IPC_CHANNELS.saveOutgoingKeepsakeDeclined,
    async (_event, payload: unknown): Promise<IpcResult> => {
      try {
        if (!validateOutgoingDeclinedRequest(payload)) {
          return { ok: false, error: "Invalid declined keepsake payload." };
        }

        const result = await repo.saveOutgoingKeepsakeDeclined(payload);
        return { ok: true, data: result };
      } catch (error) {
        console.error("[IPC] saveOutgoingKeepsakeDeclined failed:", error);
        return { ok: false, error: "Failed to save declined keepsake." };
      }
    }
  );
}

// src/renderer/ui/gift/electronGiftPersistenceApi.ts

import {
  GiftPersistenceApi,
  IncomingImageGiftPayload,
  OutgoingKeepsakeGiftPayload,
} from "./giftModal.types";

export const electronGiftPersistenceApi: GiftPersistenceApi = {
  async saveIncomingImageGift(payload: IncomingImageGiftPayload) {
    const result = await window.minariApi.gift.saveIncomingImageGift({
      imagePath: payload.imagePath,
      thumbnailPath: payload.thumbnailPath,
      userCaption: payload.userCaption,
      minariReactionLine: payload.minariReactionLine,
      visionSummaryShort: payload.visionSummaryShort,
      suggestedKeepsakeTitle: payload.suggestedKeepsakeTitle,
      createdAtIso: new Date().toISOString(),
    });

    if (!result.ok) {
      throw new Error(result.error);
    }
  },

  async saveOutgoingKeepsakeAccepted(payload: OutgoingKeepsakeGiftPayload) {
    const result = await window.minariApi.gift.saveOutgoingKeepsakeAccepted({
      keepsakeId: payload.keepsakeId,
      itemKey: payload.itemKey,
      displayName: payload.displayName,
      shortLine: payload.shortLine,
      rarityHidden: payload.rarityHidden,
      iconPath: payload.iconPath,
      description: payload.description,
      acceptedAtIso: new Date().toISOString(),
    });

    if (!result.ok) {
      throw new Error(result.error);
    }
  },

  async saveOutgoingKeepsakeDeclined(payload: OutgoingKeepsakeGiftPayload) {
    const result = await window.minariApi.gift.saveOutgoingKeepsakeDeclined({
      keepsakeId: payload.keepsakeId,
      itemKey: payload.itemKey,
      declinedAtIso: new Date().toISOString(),
    });

    if (!result.ok) {
      throw new Error(result.error);
    }
  },
};
